#Set up variable's flask

set FLASK_APP=main.py
set FLASK_DEBUG=1
set FLASK_ENV=development

#Check the data's project
gcloud config list

#set up a name's project
gcloud config set project Name_Of_Project

#Initializate project gcloud
gcloud init 

#Login to Google
gcloud auth application-default login

En caso de que queramos desplegar a githubpages (solo estaticos) podemos usar
npm i gh-pages
(se puede modificar en los scripts como deploy)
